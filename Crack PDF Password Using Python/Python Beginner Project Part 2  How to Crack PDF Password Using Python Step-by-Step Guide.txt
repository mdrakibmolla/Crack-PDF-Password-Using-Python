import pikepdf
import time

count = 0

password_file = open("password_warehouse.txt", "r")

start_time = time.time()

for password in password_file:
    new_password = password.strip("\n")
    count += 1

    try:
        pikepdf.open("March_2025.pdf", password=new_password)

        end_time = time.time()
        print(f"Password found in {round(end_time - start_time, 2)} sec")
        print(f"Password is: {new_password}")
        break

    except Exception:
        print(f"{count}: {new_password} is invalid password")
